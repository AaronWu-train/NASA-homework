  \item 使用 \texttt{qemu-img} 建立一個 16GB 的 qcow2 磁碟檔案。
\begin{minted}{bash}

    -cpu host \
    -cdrom debian-12.10.0-amd64-netinst.iso \
    -drive file=debian.qcow2 \
    -boot d \
    -net nic -net user \
    -vnc :1100
\end{minted}

  \item 根據 Docker 的官方說明，我使用以下指令安裝 Docker：
  \begin{minted}{bash}
    # Add Docker's official GPG key:
    sudo apt-get update
    sudo apt-get install ca-certificates curl
    sudo install -m 0755 -d /etc/apt/keyrings
    sudo curl -fsSL https://download.docker.com/linux/debian/gpg -o /etc/apt/keyrings/docker.asc
    sudo chmod a+r /etc/apt/keyrings/docker.asc
    # Add the repository to Apt sources:
    echo \
      "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.asc] https://download.docker.com/linux/debian \
      $(. /etc/os-release && echo "$VERSION_CODENAME") stable" | \
      sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

    sudo apt-get update
    sudo apt-get install docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin
  \end{minted}

  \begin{minted}{bash}
    sudo usermod -aG docker $USER

  \end{minted}

    git clone https://github.com/PowerDNS-Admin/PowerDNS-Admin.git

    \item 參考我先前架設的 Nginx 反向代理的配置文件
  \end{itemize}

\begin{enumerate}
  \item 在虛擬機器上安裝 Nginx。
\begin{minted}{bash}
  sudo apt install nginx
  sudo systemctl start nginx
  sudo systemctl enable nginx
\end{minted}
  \item 移除預設的 Nginx 設定檔案。
\begin{minted}{bash}
  sudo unlink /etc/nginx/sites-enabled/default
\end{minted}
  \item 新增一個 Nginx 設定檔案 \texttt{/etc/nginx/conf.d/powerdnsadmin.conf}，內容如下：
\begin{minted}{nginx}
  upstream powerdnsadmin {
      server 0.0.0.0:9191;
      keepalive 16;
  }
  server {
      listen 80 default_server;
      listen [::]:80 default_server;
      server_name b13901011.com;
      location / {
          proxy_pass http://powerdnsadmin/;
          proxy_http_version 1.1;
          proxy_set_header Host $http_host;
          proxy_set_header X-Real-IP $remote_addr;
          proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
          proxy_set_header X-Forwarded-Proto https;
      }
  }
\end{minted}
  \item 修改 Nginx 的設定檔案權限：
\begin{minted}{bash}
  sudo chown root: /etc/nginx/conf.d/powerdnsadmin.conf
  sudo chmod 644 /etc/nginx/conf.d/powerdnsadmin.conf
\end{minted}
  \item 重新啟動 Nginx 服務：
\begin{minted}{bash}
  sudo systemctl restart nginx
\end{minted}
  \item 修改 \texttt{/etc/hosts} ，新增以下的內容，使得 \texttt{b13901011.com} 能夠暫時被解析到 localhost。
\begin{minted}{bash}
  127.0.1.1       b13901011.com
\end{minted}

\subsection{Basic}
\begin{problem}{2.0.1}
  Authoritive server 指的是實際儲存一條 DNS 紀錄的最終伺服器。在 DNS 查詢過程中，會透過 root 伺服器、TLD 伺服器一路遞迴到最終的 Authoritive server 來獲取最終的 DNS 紀錄。
\end{problem}

\begin{problem}{2.0.2}
  Recursive DNS query 指的是當使用者請求時，DNS server 若不是該網址的 Authoritive server，則 DNS server 會向其他 DNS server 遞迴查詢該網址的 Authoritive server，直到獲取最終的 DNS 紀錄為止。而 Iterative DNS query 則是當使用者請求時，DNS server 會直接回傳該網址的 Authoritive server 的 IP 位址給使用者自己再去查詢，並不會遞迴查詢該網址的 Authoritive server。 Recursive DNS query 在長期來說，因為可以快取 DNS 紀錄，所以會比 Iterative DNS query 更有效率。
\end{problem}

\begin{problem}{2.1.5} 
  若 TTL 設定過高，則可能會導致伺服器將等待太久才會檢查 DNS 記錄的更新。如此一來將會導致當一個 DNS 記錄被更新時，使用者仍然會獲取舊的 DNS 記錄，使得使用者無法正確地連線到新的伺服器上。若 TTL 設定過低，則可能會導致伺服器頻繁地查詢 DNS 記錄，這樣會增加伺服器的負擔，並且可能會導致 DNS 查詢的延遲。
\end{problem}

\begin{problem}{2.1.6} 　
  \begin{enumerate}
    \item 攻擊者先對一台 DNS Server 發送一個查詢請求，而 Server 無這項紀錄，因此遞迴詢問。
    \item 攻擊者在 DNS Server 等待回應時，利用 UDP 容易被偽造的特性，發送一個偽造的回應給 DNS Server，裡面可能包含更多的 fake records，進而污染整個 Zone 的紀錄。
    \item DNS Server 收到偽造的回應後，會將這些假的 DNS 紀錄快取起來，並且在未來的查詢中回傳這些錯誤的 ip。
    \item 當使用者查詢這些 fake records 時，DNS Server 會回傳這些 fake records，而不是正確的 DNS 紀錄。使用者因此連線到假的 ip 位址。
  \end{enumerate}
\end{problem}

\subsection{Security}
\begin{problem}{2.2.1}
  倘若將 DNS recursive server 設定為 open，則將十分容易被攻擊者用來實施 DNS 放大攻擊 （一種 DDOS 攻擊）。因為攻擊者可以對我們的伺服器發送大量偽造是被攻擊目標發出的 DNS query 請求，接下來伺服器會將這些請求轉發到其他 DNS server 上，並且將回應的結果發送到被攻擊目標的伺服器上。如此一來，攻擊者就可以利用我們的伺服器來對被攻擊目標進行 DDOS 攻擊。
  另外，允許非信任裝置查詢 DNS 紀錄，也會增加 DNS cache poisoning 的風險，導致資安事件的發生。
\end{problem}

\subsection{DNS-over-TLS}


\begin{problem}{3.2.1} 　
  \begin{enumerate}[label=(\alph*)]
    \item DNS-over-TLS 是在 DNS 查詢時使用的 UDP 基礎上，多包了一層 TLS 加密的協定，這樣可以確保 DNS 查詢的內容不會被第三方竊聽或篡改。如此一來可以確保 DNS 請求不容易受到中間人攻擊的影響。
    \item  DNS-over-HTTPS (DoH) 與 DNS-over-TLS (DoT) 的比較如下：
    \begin{table}[H]
      \centering
      \begin{tabular}{lllll}
        \toprule
        & \textbf{Port} & \textbf{加密機制} & \texttt{效能} & \texttt{安全性}\\
        \midrule
        \textbf{DoT} & 853 & TLS 加密並透過 UDP 傳送 & 較快 & 有特定連接阜供管理員監控與封鎖 \\
        \textbf{DoH} & 443 & HTTPS 加密後透過 HTTP 傳送 & 較慢 & 封包隱藏在 https 當中，隱蔽性較高\\
        \bottomrule
      \end{tabular}
    \end{table}
  \end{enumerate}
\end{problem}
