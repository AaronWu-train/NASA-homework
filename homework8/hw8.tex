  \begin{minted}{bash}
    apt update
    apt install -y slapd ldap-utils ldapvi
  \end{minted}

  \begin{itemize}
    \item \texttt{suffix.ldif}：設定 LDAP 的 \texttt{olcSuffix} 為  \texttt{dc=nasa,dc=csie,dc=ntu}。
    \item \texttt{rootdn.ldif}：設定 LDAP 的 \texttt{olcRootDN}為 \texttt{cn=admin,dc=nasa,dc=csie,dc=ntu}。
  \end{itemize}
  \item 使用指令 \texttt{slappasswd} 產生密碼雜湊值，並將其寫入 \texttt{rootpw.ldif} 檔案中。
  \begin{figure}[H]
    \centering
    \includegraphics[width=0.5\textwidth]{1-1-1.png}
    \caption{使用 slappasswd 產生密碼雜湊值}
    \label{fig:slappasswd}
  \end{figure}
  \item 透過以下指令，將這三個 ldif 檔案載入 LDAP 伺服器中：
  \begin{minted}{bash}
    ldapmodify -Y EXTERNAL -H ldapi:/// -f suffix.ldif
    ldapmodify -Y EXTERNAL -H ldapi:/// -f rootdn.ldif
    ldapmodify -Y EXTERNAL -H ldapi:/// -f rootpw.ldif
  \end{minted}
  \item 按照 Lab 9 的指示，我在 \texttt{base.ldif} 檔案中，設定了 \texttt{dc=nasa,dc=csie,dc=ntu} 的節點，以及兩個 ou，並且透過以下指令套用：
  \begin{minted}{bash}
    ldapadd -D cn=admin,dc=nasa,dc=csie,dc=ntu -W	 -H ldapi:/// -f base.ldif
  \end{minted}
  \item 以下為在 Server 執行 \texttt{ldapsearch -x -b dc=nasa,dc=csie,dc=ntu} 的結果。

    \includegraphics[width=0.65\textwidth]{1-1-2.png}
    \caption{\texttt{ldapsearch} in server}
    \label{fig:ldapsearch}
  \end{figure}
\end{enumerate}

    \item 
    \item 
    \item 
  \end{itemize}
\end{mycolorbox}
以下爲我配置 LDAPS 的步驟：
\begin{enumerate}
  \item 以 root 身份登入 server。
  \item 首先，我們要先修改 server 的 hostname：
  \begin{minted}{bash}
    hostnamectl set-hostname ldap-server
  \end{minted}
  \item 接著，我們要修改 /etc/hosts 檔案，加入以下這一行：
  \begin{minted}{bash}
    127.0.1.1    ldap-server.nasa.csie.ntu    ldap-server
  \end{minted}
  \item 安裝 gnutls-bin 和 ssl-cert 套件：
  \begin{minted}{bash}
    apt-get install gnutls-bin ssl-cert
  \end{minted}
  \item 建立憑證 template，在 \texttt{/etc/ssl/templates} 目錄下創建兩個新的檔案，\texttt{ca\_server.conf} 和 \texttt{ldap\_server.conf}，並將以下內容複製到這兩個檔案中：
  \begin{minted}{bash}
    # ca_server.conf
    cn = LDAP Server CA
    ca
    cert_signing_key
  \end{minted}
  \vspace{-0.5cm}
  \begin{minted}{bash}
    # ldap_server.conf
    organization = "NTUCSIE NASA 2025"
    cn = ldap-server.nasa.csie.ntu
    tls_www_server
    encryption_key
    signing_key
    expiration_days = 3652
  \end{minted}

  \item 接下來，我們要創立 CA 的公私鑰對：
  \begin{minted}{bash}
    certtool -p --outfile /etc/ssl/private/ca_server.key
    certtool -s --load-privkey /etc/ssl/private/ca_server.key --template /etc/ssl/templates/ca_server.conf --outfile /etc/ssl/certs/ca_server.pem
  \end{minted}
  \item 利用 CA 簽署 LDAP Server 的憑證：
  \begin{minted}{bash}
    certtool -p --sec-param high --outfile /etc/ssl/private/ldap_server.key
    certtool -c --load-privkey /etc/ssl/private/ldap_server.key --load-ca-certificate /etc/ssl/certs/ca_server.pem --load-ca-privkey /etc/ssl/private/ca_server.key --template /etc/ssl/templates/ldap_server.conf --outfile /etc/ssl/certs/ldap_server.pem
  \end{minted}
  \item 修改檔案權限，使 openldap 可以讀取憑證：
  \begin{minted}{bash}
    usermod -aG ssl-cert openldap
    chown :ssl-cert /etc/ssl/private/ldap_server.key
    chmod 640 /etc/ssl/private/ldap_server.key
  \end{minted}
\end{enumerate}
