\subsection{LDAPS (LDAP over SSL)}
\begin{mycolorbox}[teal]
  \textbf{Reference:}
  \footnotesize
  \begin{itemize}
    \item \href{https://www.digitalocean.com/community/tutorials/how-to-encrypt-openldap-connections-using-starttls}{How To Encrypt OpenLDAP Connections Using STARTTLS (DigitalOcean)}
    \item \url{https://www.weithenn.org/2009/06/openldap-ssl-tls.html}
    \item \url{https://www.openldap.org/doc/admin24/tls.html}
    \item \url{https://wiki.archlinux.org/title/OpenLDAP#The_client}
  \end{itemize}
\end{mycolorbox}
以下爲我配置 LDAPS 的步驟：
\begin{enumerate}
  \item 以 root 身份登入 server。
  \item 首先，我們要先修改 server 的 hostname：
  \begin{minted}{bash}
    hostnamectl set-hostname ldap-server
  \end{minted}
  \item 接著，我們要修改 /etc/hosts 檔案，加入以下這一行：
  \begin{minted}{bash}
    127.0.1.1    ldap-server.nasa.csie.ntu    ldap-server
  \end{minted}
  \item 安裝 gnutls-bin 和 ssl-cert 套件：
  \begin{minted}{bash}
    apt-get install gnutls-bin ssl-cert
  \end{minted}
  \item 建立憑證 template，在 \texttt{/etc/ssl/templates} 目錄下創建兩個新的檔案，\texttt{ca\_server.conf} 和 \texttt{ldap\_server.conf}，並將以下內容複製到這兩個檔案中：
  \begin{minted}{bash}
    # ca_server.conf
    cn = LDAP Server CA
    ca
    cert_signing_key
  \end{minted}
  \vspace{-0.5cm}
  \begin{minted}{bash}
    # ldap_server.conf
    organization = "NTUCSIE NASA 2025"
    cn = ldap-server.nasa.csie.ntu
    tls_www_server
    encryption_key
    signing_key
    expiration_days = 3652
  \end{minted}

  \item 接下來，我們要創立 CA 的公私鑰對：
  \begin{minted}{bash}
    certtool -p --outfile /etc/ssl/private/ca_server.key
    certtool -s --load-privkey /etc/ssl/private/ca_server.key --template /etc/ssl/templates/ca_server.conf --outfile /etc/ssl/certs/ca_server.pem
  \end{minted}
  \item 利用 CA 簽署 LDAP Server 的憑證：
  \begin{minted}{bash}
    certtool -p --sec-param high --outfile /etc/ssl/private/ldap_server.key
    certtool -c --load-privkey /etc/ssl/private/ldap_server.key --load-ca-certificate /etc/ssl/certs/ca_server.pem --load-ca-privkey /etc/ssl/private/ca_server.key --template /etc/ssl/templates/ldap_server.conf --outfile /etc/ssl/certs/ldap_server.pem
  \end{minted}
  \item 修改檔案權限，使 openldap 可以讀取憑證：
  \begin{minted}{bash}
    usermod -aG ssl-cert openldap
    chown :ssl-cert /etc/ssl/private/ldap_server.key
    chmod 640 /etc/ssl/private/ldap_server.key
  \end{minted}

  \item 建立 \texttt{addcerts.ldif} 檔案，新增以下內容：
  \begin{minted}{ldif}
    # addcerts.ldif
    dn: cn=config
    changetype: modify
    add: olcTLSCACertificateFile
    olcTLSCACertificateFile: /etc/ssl/certs/ca_server.pem
    -
    add: olcTLSCertificateFile
    olcTLSCertificateFile: /etc/ssl/certs/ldap_server.pem
    -
    add: olcTLSCertificateKeyFile
    olcTLSCertificateKeyFile: /etc/ssl/private/ldap_server.key
  \end{minted}
  \item 以 \texttt{ldapmodify -H ldapi:// -Y EXTERNAL -f addcerts.ldif} 指令將憑證載入 LDAP Server。

  \item 修改 \texttt{/etc/default/slapd} 檔案，將 \texttt{SLAPD\_SERVICES} 的改為：
  \begin{minted}{bash}
    SLAPD_SERVICES="ldap:/// ldapi:/// ldaps:///"
  \end{minted}

  \item 複製一份公鑰：\texttt{cp /etc/ssl/certs/ca\_server.pem /etc/ldap/ca\_certs.pem}
  \item 修改 \texttt{/etc/ldap/ldap.conf} 檔案，修改 \texttt{TLS\_CACERT}：
  \begin{minted}{bash}
    TLS_CACERT /etc/ldap/ca_certs.pem
  \end{minted}

  \item 以 \texttt{systemctl restart slapd} 指令重啟 LDAP Server。
  \item 以 \texttt{reboot} 重新開機 server。
  \item 使用以下指令測試憑證，發現可以成功連線，顯示憑證鏈：
  \begin{minted}{bash}
    openssl s_client -starttls ldap -connect ldap-server.nasa.csie.ntu:389 -showcerts
  \end{minted}
  \item 使用題目給予的指令，我發現我可以成功以 StartTLS 或 SSL 連線到 Server。
  \begin{mycolorbox}[green]
    \textbf{Answer:}
    \begin{figure}[H]
      \centering
      \includegraphics[width=0.75\textwidth]{2-1-1.png}
      \caption{使用 StartTLS 連線 (\texttt{ldapsearch -x -ZZ -b dc=nasa,dc=csie,dc=ntu})}
      \label{fig:starttls}
    \end{figure}
    \begin{figure}[H]
      \centering
      \includegraphics[width=0.75\textwidth]{2-1-2.png}
      \caption{使用 SSL 連線 (\texttt{ldapsearch -x -H ldaps:/// -b dc=nasa,dc=csie,dc=ntu})}
      \label{fig:sssl}
    \end{figure}
  \end{mycolorbox}
  \item 為了只允許 SSL 連線，我建立了 \texttt{onlyssl.ldif}：
  \begin{minted}{ldif}
    # onlyssl.ldif
    dn: cn=config
    changetype: modify
    replace: olcSecurity
    olcSecurity: ssf=128 update_ssf=128 minssf=128 tls=1
  \end{minted}
  並且以 \texttt{ldapmodify -H ldapi:// -Y EXTERNAL -f onlyssl.ldif} 指令載入。
  \item 接下來，將 server 上的 \texttt{/etc/ssl/certs/ca\_server.pem} 公鑰檔案複製到 client 上，放置到 \texttt{/etc/ca-certificates/trust-source/anchors/ca-nasa-ldap.crt} 中。
  \item 執行以下指令更新憑證，此指令會將憑證複製到\texttt{/etc/ssl/certs/LDAP\_Server\_CA.pem} 中，並加入系統的信任列表：
  \begin{minted}{bash}
    update-ca-trust extract
  \end{minted}
  \item 修改 client 的 \texttt{/etc/hosts} 檔案，加入以下這一行：
  \begin{minted}{bash}
    192.168.128.225 ldap-server.nasa.csie.ntu
  \end{minted}
  \item 修改 client 上的 \texttt{/etc/openldap/ldap.conf} 檔案為以下：
  \begin{minted}{bash}
    URI     ldaps://ldap-server.nasa.csie.ntu
  \end{minted}
  \item 執行 \texttt{ldapsearch -x -H -b dc=nasa,dc=csie,dc=ntu} 指令，發現可以用 ssl 成功連線，如下圖所示：
  \begin{mycolorbox}[green]
    \textbf{Answer:}
    \begin{figure}[H]
      \centering
      \includegraphics[width=0.8\textwidth]{2-1-3.png}
      \caption{使用 SSL 連線 (\texttt{ldapsearch -x -H ldaps:/// -b dc=nasa,dc=csie,dc=ntu})}
      \label{fig:sssl-client}
    \end{figure}  
  \end{mycolorbox}

  \item 以非 ssl 的方式連線會失敗：
  \begin{minted}{bash}
    ldapsearch -x -H ldap://192.168.128.225 -b dc=nasa,dc=csie,dc=ntu
  \end{minted}
  \begin{mycolorbox}[green]
    \textbf{Answer:}
    \begin{figure}[H]
      \centering
      \includegraphics[width=\textwidth]{2-1-4.png}
      \caption{使用非 SSL 連線被阻擋}
      \label{fig:sssl-client-fail}
    \end{figure}
  \end{mycolorbox}
\end{enumerate}

\begin{enumerate}
  \item 首先，我在 client 上安裝 SSSD 和 sudo 套件：
  \begin{minted}{bash}
    pacman -S sssd sudo
  \end{minted}
  \item 接著，編輯 \texttt{/etc/sssd/sssd.conf} 檔案，加入以下內容：
  \begin{minted}{bash}
    [sssd]
    domains = nasa.csie.ntu
    services = nss, pam

    [domain/nasa.csie.ntu]
    id_provider        = ldap
    auth_provider      = ldap
    chpass_provider    = ldap
    ldap_uri           = ldaps://ldap-server.nasa.csie.ntu
    ldap_tls_reqcert   = demand
    ldap_search_base   = dc=nasa,dc=csie,dc=ntu
    cache_credentials  = true
    fallback_homedir   = /home/%u
  \end{minted}
  \item 完成後，將檔案的權限設為 600：
  \begin{minted}{bash}
    chmod 600 /etc/sssd/sssd.conf
  \end{minted}
  \item 接著，設定 PAM，允許自動新增使用者資料夾。編輯 \texttt{/etc/pam.d/system-auth} 檔案，加入以下內容：
  \begin{minted}{bash}
    auth sufficient pam_sss.so forward_pass # 加在最前面
    account [default=bad success=ok user_unknown=ignore authinfo_unavail=ignore] pam_sss.so # 加在 account 區段最前面
    password sufficient pam_sss.so # 加在 password 區段最前面
    session required pam_mkhomedir.so skel=/etc/skel umask=0022
    session optional pam_sss.so
  \end{minted}
  \item 接著，啟用 SSSD 服務：
  \begin{minted}{bash}
    systemctl enable sssd
    systemctl start sssd
  \end{minted}
  \item 編輯 Client 的 \texttt{/etc/sudoers.d/ta} 檔案，加入以下內容：
  \begin{minted}{bash}
    %ta ALL=(ALL) ALL
  \end{minted}
  然後以指令 \texttt{chmod 440 /etc/sudoers.d/ta} 設定權限。
  \item 編輯 \texttt{/etc/nsswitch.conf} 檔案，將 \texttt{passwd}、\texttt{group} 和 \texttt{shadow} 的值加上 \texttt{sss}：
  \begin{minted}{bash}
    passwd: files sss systemd
    group: files sss [SUCCESS=merge] systemd
    shadow: files sss systemd
  \end{minted}
  \item 修改 \texttt{/etc/ssh/sshd\_config} 檔案，加入以下修改：
  \begin{minted}{bash}
    UsePAM yes
    PasswordAuthentication yes
    ChallengeResponseAuthentication no    
  \end{minted}
  \item 接著，以 \texttt{systemctl restart sshd} 和 \texttt{systemctl restart sssd}重啟 sshd 和 sssd 服務。
  \item 回到 server 上，新增檔案 \texttt{addgroup.ldif} 用來加入兩個群組，並且用以下指令套用：
  \begin{minted}{bash}
    ldapadd -x -H ldaps:/// -D "cn=admin,dc=nasa,dc=csie,dc=ntu" -W -f addgroup.ldif
  \end{minted}
  \item 使用指令 \texttt{slappasswd} 產生兩個密碼雜湊值，作為新增使用者的密碼。
  \item 接著，新增檔案 \texttt{adduser.ldif} 用來新增兩個使用者，分別為 \texttt{ta-1} 和 \texttt{student-1}，並用以下指令套用：

  \end{minted}
  \item 分別以 ssh 連線到 client 上，如以下結果所示：

\subsection{ACL (Access Control Lists)}
\begin{mycolorbox}[teal]
  \textbf{Reference:}
  \footnotesize
  \begin{itemize}
    \item \url{https://www.weithenn.org/2009/06/openldap-server-install.html}
    \item \url{https://www.openldap.org/doc/admin24/access-control.html}
    \item \url{https://uit.stanford.edu/service/directory/aclexamples}
  \end{itemize}
\end{mycolorbox}
首先，我建立了一個 \texttt{acl.ldif} 檔案，並且加入以下內容：
\begin{minted}{ldif}
  dn: olcDatabase={1}mdb,cn=config
  changetype: modify
  replace: olcAccess
  olcAccess: {0}to attrs=userPassword by self write by anonymous auth by * none
  olcAccess: {1}to attrs=cn,uid,gidNumber,homeDirectory by self read by users read by anonymous read
  olcAccess: {2}to dn.subtree=ou=people,dc=nasa,dc=csie,dc=ntu by self write by users read by anonymous read
  olcAccess: {3}to * by self write by users read by anonymous read
\end{minted}
然後我利用以下指令將 ACL 套用到 LDAP Server：
\begin{minted}{bash}
  ldapmodify -Y EXTERNAL -H ldapi:/// -f acl.ldif
\end{minted}
